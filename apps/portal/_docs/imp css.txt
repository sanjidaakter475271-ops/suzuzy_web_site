      const interval = setInterval(() => {
        setProgress(prev => {

// Stats Card Component
function StatsCard({ 
  title, 
  value, 
  change, 
  changeType, 
  icon: Icon, 
  color 
}: {
  title: string;
  value: string;
  change: string;
  changeType: 'increase' | 'decrease';
  icon: React.ElementType;
  color: 'blue' | 'green' | 'amber' | 'red';
}) {
  const colors = {
    blue: 'from-blue-500 to-blue-600',
    green: 'from-green-500 to-green-600',
    amber: 'from-amber-500 to-amber-600',
    red: 'from-red-500 to-red-600',
  };

  const bgColors = {
    blue: 'bg-blue-50',
    green: 'bg-green-50',
    amber: 'bg-amber-50',
    red: 'bg-red-50',
  };

  return (
    <motion.div
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      whileHover={{ y: -4, boxShadow: '0 12px 24px -8px rgba(0,0,0,0.15)' }}
      className="bg-white rounded-xl border border-slate-200 p-5 relative overflow-hidden group cursor-pointer"
    >
      {/* Top Gradient Line */}
      <div className={`absolute top-0 left-0 right-0 h-1 bg-gradient-to-r ${colors[color]}`} />
      
      <div className="flex items-start justify-between">
        <div>
          <p className="text-sm font-medium text-slate-500 mb-1">{title}</p>
          <motion.h3
            initial={{ scale: 0.5 }}
            animate={{ scale: 1 }}
            className="text-2xl font-bold text-slate-900"
          >
            {value}
          </motion.h3>
          
          {/* Change Indicator */}
          <div className="flex items-center gap-1 mt-2">
            {changeType === 'increase' ? (
              <TrendingUp className="w-4 h-4 text-green-500" />
            ) : (
              <TrendingDown className="w-4 h-4 text-red-500" />
            )}
            <span className={`text-sm font-medium ${
              changeType === 'increase' ? 'text-green-600' : 'text-red-600'
            }`}>
              {change}
            </span>
            <span className="text-sm text-slate-400">vs last month</span>
          </div>
        </div>

        {/* Icon */}
        <div className={`${bgColors[color]} p-3 rounded-xl group-hover:scale-110 transition-transform`}>
          <Icon className={`w-6 h-6 bg-gradient-to-r ${colors[color]} bg-clip-text text-transparent`} />
        </div>
      </div>

      {/* Hover Arrow */}
      <motion.div
        initial={{ opacity: 0, x: -10 }}
        whileHover={{ opacity: 1, x: 0 }}
        className="absolute bottom-4 right-4 text-slate-400"
      >
        <ArrowRight className="w-5 h-5" />
      </motion.div>
    </motion.div>
  );
}

// Alert Item Component
function AlertItem({ 
  icon: Icon, 
  title, 
  count, 
  severity 
}: {
  icon: React.ElementType;
  title: string;
  count: number;
  severity: 'warning' | 'error' | 'info';
}) {
  const severityStyles = {
    warning: 'bg-amber-50 text-amber-600 border-amber-200',
    error: 'bg-red-50 text-red-600 border-red-200',
    info: 'bg-blue-50 text-blue-600 border-blue-200',
  };

  return (
    <motion.div
      initial={{ opacity: 0, x: -20 }}
      animate={{ opacity: 1, x: 0 }}
      whileHover={{ x: 4 }}
      className={`flex items-center gap-3 p-3 rounded-lg border ${severityStyles[severity]} cursor-pointer`}
    >
      <Icon className="w-5 h-5" />
      <span className="flex-1 font-medium">{title}</span>
      <span className="bg-white px-2 py-0.5 rounded-full text-sm font-bold">
        {count}
      </span>
    </motion.div>
  );
}

// ===========================================
// PAGE LOADER CONTROLLER
// ===========================================

class PageLoader {
  constructor() {
    this.loader = document.getElementById('pageLoader');
    this.progress = document.querySelector('.loader-progress__bar');
    this.message = document.querySelector('.loader-message');
    this.messages = [
      'Starting engines...',
      'Loading bikes...',
      'Checking inventory...',
      'Almost there...',
      'Ready to ride!'
    ];
    this.currentProgress = 0;
  }

  init() {
    this.simulateProgress();
    window.addEventListener('load', () => this.complete());
  }

  simulateProgress() {
    const interval = setInterval(() => {
      if (this.currentProgress < 90) {
        this.currentProgress += Math.random() * 15;
        this.updateProgress(Math.min(this.currentProgress, 90));
      } else {
        clearInterval(interval);
      }
    }, 300);
  }

  updateProgress(value) {
    this.progress.style.width = `${value}%`;
    
    // Update message based on progress
    const messageIndex = Math.floor((value / 100) * this.messages.length);
    this.message.textContent = this.messages[Math.min(messageIndex, this.messages.length - 1)];
  }

  complete() {
    this.updateProgress(100);
    this.message.textContent = 'Ready to ride!';
    
    setTimeout(() => {
      this.loader.classList.add('loaded');
      document.body.classList.add('loaded');
      
      // Remove from DOM after animation
      setTimeout(() => {
        this.loader.remove();
      }, 500);
    }, 500);
  }
}

  pageLoader.init();

const Loading = {
  // Show skeleton for a container
  showSkeleton(container, type = 'card', count = 1) {
    const skeletons = [];
    
    for (let i = 0; i < count; i++) {
      const skeleton = document.createElement('div');
      skeleton.className = `skeleton-${type}`;
      skeleton.innerHTML = this.getSkeletonHTML(type);
      skeletons.push(skeleton);
    }
    
    container.innerHTML = '';
    skeletons.forEach(s => container.appendChild(s));
  },

  getSkeletonHTML(type) {
    const templates = {
      card: `
        <div class="product-card-skeleton">
          <div class="product-card-skeleton__image skeleton"></div>
          <div class="product-card-skeleton__content">
            <div class="skeleton skeleton--text-short"></div>
            <div class="skeleton skeleton--title"></div>
            <div class="skeleton skeleton--text"></div>
            <div class="skeleton skeleton--button"></div>
          </div>
        </div>
      `,
      table: `
        <div class="skeleton skeleton--text" style="margin-bottom: 16px;"></div>
        <div class="skeleton skeleton--text"></div>
        <div class="skeleton skeleton--text-short"></div>
      `,
      profile: `
        <div style="display: flex; gap: 16px; align-items: center;">
          <div class="skeleton skeleton--avatar"></div>
          <div style="flex: 1;">
            <div class="skeleton skeleton--text-short"></div>
            <div class="skeleton skeleton--text"></div>
          </div>
        </div>
      `
    };
    
    return templates[type] || templates.card;
  },

  // Show spinner button
  buttonLoading(button, loading = true) {
    if (loading) {
      button.disabled = true;
      button.dataset.originalText = button.innerHTML;
      button.innerHTML = `
        <span class="spinner spinner--sm"></span>
        <span>Loading...</span>
      `;
    } else {
      button.disabled = false;
      button.innerHTML = button.dataset.originalText;
    }
  },

  // Show table overlay
  tableLoading(table, loading = true) {
    if (loading) {
      table.classList.add('table-loading');
      const spinner = document.createElement('div');
      spinner.className = 'table-loading__spinner';
      spinner.innerHTML = '<div class="spinner spinner--lg"></div>';
      table.appendChild(spinner);
    } else {
      table.classList.remove('table-loading');
      const spinner = table.querySelector('.table-loading__spinner');
      if (spinner) spinner.remove();
    }
  }
};
